'use strict';

const express = require('express');
const cors = require('cors');
const mongoose = require('mongoose');
require('dotenv').config();

const app = express();
app.use(cors());
app.use(express.urlencoded({ extended: false }));
app.use(express.json());

const MONGO_URI = process.env.MONGO_URI;
mongoose
  .connect(MONGO_URI, { dbName: 'exercise_tracker' })
  .then(() => console.log('✅ Connected to MongoDB'))
  .catch(err => console.error('❌ Mongo error:', err.message));

const userSchema = new mongoose.Schema({
  username: { type: String, required: true }
});
const User = mongoose.model('User', userSchema);

const exerciseSchema = new mongoose.Schema({
  userId: { type: mongoose.Schema.Types.ObjectId, ref: 'User', required: true },
  description: { type: String, required: true },
  duration: { type: Number, required: true },
  date: { type: Date, default: Date.now }
});
const Exercise = mongoose.model('Exercise', exerciseSchema);

// Create user
app.post('/api/users', async (req, res) => {
  try {
    const { username } = req.body;
    const user = await User.create({ username });
    res.json({ username: user.username, _id: user._id });
  } catch (e) {
    res.status(500).json({ error: 'failed to create user' });
  }
});

// List users
app.get('/api/users', async (_req, res) => {
  const users = await User.find({}, { __v: 0 });
  res.json(users);
});

// Add exercise
app.post('/api/users/:_id/exercises', async (req, res) => {
  try {
    const { _id } = req.params;
    const { description, duration, date } = req.body;

    const user = await User.findById(_id);
    if (!user) return res.status(400).json({ error: 'unknown user' });

    const when = date ? new Date(date) : new Date();
    if (isNaN(when)) return res.json({ error: 'invalid date' });

    const exercise = await Exercise.create({
      userId: user._id,
      description,
      duration: Number(duration),
      date: when
    });

    res.json({
      username: user.username,
      description: exercise.description,
      duration: exercise.duration,
      date: exercise.date.toDateString(),
      _id: user._id
    });
  } catch (e) {
    res.status(500).json({ error: 'failed to add exercise' });
  }
});

// Get logs
app.get('/api/users/:_id/logs', async (req, res) => {
  try {
    const { _id } = req.params;
    const { from, to, limit } = req.query;

    const user = await User.findById(_id);
    if (!user) return res.status(400).json({ error: 'unknown user' });

    const q = { userId: user._id };
    if (from || to) {
      q.date = {};
      if (from) q.date.$gte = new Date(from);
      if (to) q.date.$lte = new Date(to);
    }

    let query = Exercise.find(q).select({ __v: 0, userId: 0 }).sort({ date: 1 });
    if (limit) query = query.limit(Number(limit));

    const docs = await query.exec();

    res.json({
      username: user.username,
      count: docs.length,
      _id: user._id,
      log: docs.map(d => ({
        description: d.description,
        duration: d.duration,
        date: d.date.toDateString()
      }))
    });
  } catch (e) {
    res.status(500).json({ error: 'failed to get logs' });
  }
});

// Health
app.get('/', (_req, res) => res.send('Exercise Tracker API is running'));

const port = process.env.PORT || 3000;
app.listen(port, () => console.log('✅ Server listening on ' + port));
